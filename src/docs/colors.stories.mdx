# Colors
<br /><br />
#### All colors are defined in the theme, and can be used anywhere by their key name.

```jsx
import { Box } from '../core/components/Box/Box';
import { StyleSystemProps } from '../core/styling/config/styled-system.types';

// you can use them directly to any color props 
<Box bg="gray1" />
<Box color="gray4" />
...

// or... set them in a StyleSystemProps object to be passed as props to any Box component 
const boxStyles: StyleSystemProps = {
  borderColor: 'yellow1',
  ...
};
<Box {...boxStyles} />
```
```jsx
// optionally, you can set raw css values. 
// however, this is not recommended, as we want to keep the theme consistent
<Box color="red" />
<Box color="#451225" />
```
```jsx
// responsive values are also supported
<Box color={['red1', null, 'blue1', 'green1', 'gray1']} />
<Box color={{ sm: 'red1', md: 'blue1', lg: 'green1', xl:'gray1' }} />

const boxStyles: StyleSystemProps = {
  backgroundColor: {
    _: 'red1',
    md: 'blue1',
    lg: 'green1',
    xl: 'gray1'
  }
  ...
};
<Box {...boxStyles} />
```
